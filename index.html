const Pi = window.Pi; Pi.init({ version: "2.0", sandbox: true }); const loginBtn = document.getElementById("loginBtn"); const payBtn = document.getElementById("payBtn"); const usernameBox = document.getElementById("username"); const statusBox = document.getElementById("status"); function status(text){ statusBox.innerText = text; } let currentUser = null; loginBtn.addEventListener("click", async () => { try{ status("Logging in..."); const auth = await Pi.authenticate( ["username","payments"], { onIncompletePaymentFound: async payment => { await fetch("/api/complete",{ method:"POST", headers:{ "Content-Type":"application/json" }, body:JSON.stringify({ paymentId: payment.identifier }) }); } } ); currentUser = auth.user; usernameBox.innerText = currentUser.username; loginBtn.disabled = true; payBtn.disabled = false; status("Login successful"); }catch(err){ console.error(err); status("Login failed"); } });
